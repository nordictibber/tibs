  <script>

    const questionData = [

      { id: "high_avg_price", question: "Betaler du mer enn 50 øre/kWh i snitt i løpet av året?" },

      { id: "high_usage", question: "Bruker husholdningen din mer enn 16 000 kWh i året?" },

      { id: "south_norway", question: "Bor du i Sør-Norge?" },

      { id: "wants_stability", question: "Ønsker du forutsigbare strømregninger framfor svingende priser?" },

      { id: "no_time", question: "Har du lite tid eller lyst til å følge med på strømprisene daglig?" }

    ];


    let current = 0;

    let yesCount = 0;

    const quizEl = document.getElementById("quiz");

    const resultEl = document.getElementById("result");


    function showQuestion() {

      if (current < questionData.length) {

        const currentQuestion = questionData[current];

        const questionHTML = `

          <div class="question" id="q${current}">

            <p>${currentQuestion.question}</p>

            <div class="buttons">

              <button onclick="answer('yes')">Ja</button>

              <button onclick="answer('no')">Nei</button>

            </div>

          </div>

        `;


        quizEl.innerHTML = questionHTML;


        // Fade in

        requestAnimationFrame(() => {

          document.getElementById(`q${current}`).classList.add('show');

        });


      } else {

        quizEl.innerHTML = '';

        resultEl.textContent = yesCount >= 3

          ? "✅ JA – du burde gå for Norgespris!"

          : "❌ NEI – det lønner seg trolig mer med spotpris og smarte løsninger.";

      }

    }


    function answer(choice) {

      if (choice === 'yes') yesCount++;


      const currentQuestionEl = document.getElementById(`q${current}`);

      currentQuestionEl.classList.remove('show');


      setTimeout(() => {

        current++;

        showQuestion();

      }, 400); // wait for fade-out

    }


    showQuestion();

  </script>
